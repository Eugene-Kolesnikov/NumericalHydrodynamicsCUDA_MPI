22 void initializeField(Cell* field, size_t N_X, size_t N_Y)
23 {
24  dx = xsize / (N_X-1); // horizontal grid step, m
25  dy = ysize / (N_Y-1); // vertical grid step
26 
27  float* x = new float[N_X]; // horizontal coordinates of grid points, m
28  float* y = new float[N_Y]; // vertical coordinates of grid points, m
29 
30  // initialize x,y
31  for(int i = 0; i < N_X; ++i) {
32  x[i] = i*dx;
33  }
34  for(int j = 0; j < N_Y; ++j) {
35  y[j] = j*dy;
36  }
37 
38  R = new float[N_X*N_Y];
39  float* RHO = new float[N_X*N_Y]; // Density (temporary matrix)
40  double G = 6.672e-11;
41  // compose density array
42  float rp = 6000000; // planetary radius, m
43  // radius for BC (FI=0)
44  float minsize = (xsize > ysize ? ysize : xsize);
45  float mindr = (dx > dy ? dy : dx);
46  float rBC = minsize / 2.0f - mindr / 2.0f;
47 
48  for(int j = 0; j < N_Y; ++j) {
49  for(int i = 0; i < N_X; ++i) {
50  float offset_x = x[i] - xsize / 2.0f;
51  float offset_y = y[j] - ysize / 2.0f;
52  float rcurrent = sqrt((offset_x*offset_x + offset_y*offset_y)); // distance from the center of the model
53  if(rcurrent > rp) {
54  RHO[j*N_X+i] = 0; // outside of the planet
55  } else {
56  RHO[j*N_X+i] = 5500; // inside the planet
57  }
58  // initialization of R (result vector)
59  if(rcurrent > rBC) { // boundary condition
60  R[j*N_X+i] = 0;
61  } else {
62  R[j*N_X+i] = 2.0f / 3.0f * 4.0f * M_PI * G * RHO[j*N_X+i];
63  }
64  }
65  }
66 
67  // first guess
68  for(int i = 0; i < N_X * N_Y; ++i) {
69  field[i].r = 0.0f;
70  }
71 
72  delete[] RHO;
73  delete[] x;
74  delete[] y;
75 }

77 void GaussSeidelSolver(Cell* field, size_t N_X, size_t N_Y)
78 {
79  size_t Iter = 1000;
80  float theta = 1.5f; // relaxation parameter
81  for(int I = 0; I < Iter; ++I)
82  {
83  for(int j = 0; j < N_Y; ++j)
84  {
85  for(int i = 0; i < N_X; ++i)
86  {
87  if(i == 0 || j == 0 || i == (N_X-1) || j == (N_Y-1)) {
88  field[j*N_X+i].r = 0.0f;
89  } else {
90  field[j*N_X+i].r = field[j*N_X+i].r + theta *
91  ((field[j*N_X+i+1].r - 2*field[j*N_X+i].r + field[j*N_X+i-1].r) / (dx*dx)
92  + (field[(j+1)*N_X+i].r - 2*field[j*N_X+i].r + field[(j-1)*N_X+i].r) / (dy*dy)
93  - R[j*N_X+i]) / (2.0f/(dx*dx) + 2.0f/(dy*dy));
94  }
95  }
96  }
97  }

104 
105  for(int i = 0; i < N_X * N_Y; ++i) {

107  minval = field[i].r;
108  }
109  }
110  for(int i = 0; i < N_X * N_Y; ++i) {
111  field[i].r = (field[i].r - minval);
112  }
113  for(int i = 0; i < N_X * N_Y; ++i) {
114  if(field[i].r > maxval) {
115  maxval = field[i].r;
116  }
117  }
118  for(int i = 0; i < N_X * N_Y; ++i) {
119  field[i].r = field[i].r / maxval;
